153 /* Text is the actual textual contents of span. */
154 typedef enum MD_TEXTTYPE
155 {
156 /* Normal text. */
157 MD_TEXT_NORMAL = 0,
158
159 /* NULL character. CommonMark requires replacing NULL character with
160 * the replacement char U+FFFD, so this allows caller to do that easily. */
161 MD_TEXT_NULLCHAR,
162
163 /* Line breaks.
164 * Note these are not sent from blocks with verbatim output (MD_BLOCK_CODE
165 * or MD_BLOCK_HTML). In such cases, '\n' is part of the text itself. */
166 MD_TEXT_BR, /* <br> (hard break) */
167 MD_TEXT_SOFTBR, /* '\n' in source text where it is not semantically meaningful (soft break) */
168
169 /* Entity.
170 * (a) Named entity, e.g. &nbsp;
171 * (Note MD4C does not have a list of known entities.
172 * Anything matching the regexp /&[A-Za-z][A-Za-z0-9]{1,47};/ is
173 * treated as a named entity.)
174 * (b) Numerical entity, e.g. &#1234;
175 * (c) Hexadecimal entity, e.g. &#x12AB;
176 *
177 * As MD4C is mostly encoding agnostic, application gets the verbatim
178 * entity text into the MD_PARSER::text_callback(). */
179 MD_TEXT_ENTITY,
180
181 /* Text in a code block (inside MD_BLOCK_CODE) or inlined code (`code`).
182 * If it is inside MD_BLOCK_CODE, it includes spaces for indentation and
183 * '\n' for new lines. MD_TEXT_BR and MD_TEXT_SOFTBR are not sent for this
184 * kind of text. */
185 MD_TEXT_CODE,
186
187 /* Text is a raw HTML. If it is contents of a raw HTML block (i.e. not
188 * an inline raw HTML), then MD_TEXT_BR and MD_TEXT_SOFTBR are not used.
189 * The text contains verbatim '\n' for the new lines. */
190 MD_TEXT_HTML,
191
192 /* Text is inside an equation. This is processed the same way as inlined code
193 * spans (`code`). */
194 MD_TEXT_LATEXMATH
195 } MD_TEXTTYPE;

206 /* String attribute.
207 *
208 * This wraps strings which are outside of a normal text flow and which are
209 * propagated within various detailed structures, but which still may contain
210 * string portions of different types like e.g. entities.
211 *
212 * So, for example, lets consider this image:
213 *
214 * ![image alt text](http://example.org/image.png 'foo &quot; bar')
215 *
216 * The image alt text is propagated as a normal text via the MD_PARSER::text()
217 * callback. However, the image title ('foo &quot; bar') is propagated as
218 * MD_ATTRIBUTE in MD_SPAN_IMG_DETAIL::title.
219 *
220 * Then the attribute MD_SPAN_IMG_DETAIL::title shall provide the following:
221 * -- [0]: "foo " (substr_types[0] == MD_TEXT_NORMAL; substr_offsets[0] == 0)
222 * -- [1]: "&quot;" (substr_types[1] == MD_TEXT_ENTITY; substr_offsets[1] == 4)
223 * -- [2]: " bar" (substr_types[2] == MD_TEXT_NORMAL; substr_offsets[2] == 10)
224 * -- [3]: (n/a) (n/a ; substr_offsets[3] == 14)
225 *
226 * Note that these invariants are always guaranteed:
227 * -- substr_offsets[0] == 0
228 * -- substr_offsets[LAST+1] == size
229 * -- Currently, only MD_TEXT_NORMAL, MD_TEXT_ENTITY, MD_TEXT_NULLCHAR
230 * substrings can appear. This could change only of the specification
231 * changes.
232 */
233 typedef struct MD_ATTRIBUTE
234 {
235 const MD_CHAR* text;
236 MD_SIZE size;
237 const MD_TEXTTYPE* substr_types;
238 const MD_OFFSET* substr_offsets;
239 } MD_ATTRIBUTE;

313 /* Flags specifying extensions/deviations from CommonMark specification.
314 *
315 * By default (when MD_PARSER::flags == 0), we follow CommonMark specification.
316 * The following flags may allow some extensions or deviations from it.
317 */
318#define MD_FLAG_COLLAPSEWHITESPACE 0x0001 /* In MD_TEXT_NORMAL, collapse non-trivial whitespace into single ' ' */
319#define MD_FLAG_PERMISSIVEATXHEADERS 0x0002 /* Do not require space in ATX headers ( ###header ) */
320#define MD_FLAG_PERMISSIVEURLAUTOLINKS 0x0004 /* Recognize URLs as autolinks even without '<', '>' */
321#define MD_FLAG_PERMISSIVEEMAILAUTOLINKS 0x0008 /* Recognize e-mails as autolinks even without '<', '>' and 'mailto:' */
322#define MD_FLAG_NOINDENTEDCODEBLOCKS 0x0010 /* Disable indented code blocks. (Only fenced code works.) */
323#define MD_FLAG_NOHTMLBLOCKS 0x0020 /* Disable raw HTML blocks. */
324#define MD_FLAG_NOHTMLSPANS 0x0040 /* Disable raw HTML (inline). */
325#define MD_FLAG_TABLES 0x0100 /* Enable tables extension. */
326#define MD_FLAG_STRIKETHROUGH 0x0200 /* Enable strikethrough extension. */
327#define MD_FLAG_PERMISSIVEWWWAUTOLINKS 0x0400 /* Enable WWW autolinks (even without any scheme prefix, if they begin with 'www.') */
328#define MD_FLAG_TASKLISTS 0x0800 /* Enable task list extension. */
329#define MD_FLAG_LATEXMATHSPANS 0x1000 /* Enable $ and $$ containing LaTeX equations. */
330#define MD_FLAG_WIKILINKS 0x2000 /* Enable wiki links extension. */
331#define MD_FLAG_UNDERLINE 0x4000 /* Enable underline extension (and disables '_' for normal emphasis). */
332#define MD_FLAG_HARD_SOFT_BREAKS 0x8000 /* Force all soft breaks to act as hard breaks. */
333
334#define MD_FLAG_PERMISSIVEAUTOLINKS (MD_FLAG_PERMISSIVEEMAILAUTOLINKS | MD_FLAG_PERMISSIVEURLAUTOLINKS | MD_FLAG_PERMISSIVEWWWAUTOLINKS)
335#define MD_FLAG_NOHTML (MD_FLAG_NOHTMLBLOCKS | MD_FLAG_NOHTMLSPANS)
336
337 /* Convenient sets of flags corresponding to well-known Markdown dialects.
338 *
339 * Note we may only support subset of features of the referred dialect.
340 * The constant just enables those extensions which bring us as close as
341 * possible given what features we implement.
342 *
343 * ABI compatibility note: Meaning of these can change in time as new
344 * extensions, bringing the dialect closer to the original, are implemented.
345 */
346#define MD_DIALECT_COMMONMARK 0
347#define MD_DIALECT_GITHUB (MD_FLAG_PERMISSIVEAUTOLINKS | MD_FLAG_TABLES | MD_FLAG_STRIKETHROUGH | MD_FLAG_TASKLISTS)

349 /* Parser structure.
350 */
351 typedef struct MD_PARSER
352 {
353 /* Reserved. Set to zero.
354 */
355 unsigned abi_version;
356
357 /* Dialect options. Bitmask of MD_FLAG_xxxx values.
358 */
359 unsigned flags;
360
361 /* Caller-provided rendering callbacks.
362 *
363 * For some block/span types, more detailed information is provided in a
364 * type-specific structure pointed by the argument 'detail'.
365 *
366 * The last argument of all callbacks, 'userdata', is just propagated from
367 * md_parse() and is available for any use by the application.
368 *
369 * Note any strings provided to the callbacks as their arguments or as
370 * members of any detail structure are generally not zero-terminated.
371 * Application has to take the respective size information into account.
372 *
373 * Any rendering callback may abort further parsing of the document by
374 * returning non-zero.
375 */
376 int (*enter_block)(MD_BLOCKTYPE /*type*/, void* /*detail*/, void* /*userdata*/);
377 int (*leave_block)(MD_BLOCKTYPE /*type*/, void* /*detail*/, void* /*userdata*/);
378
379 int (*enter_span)(MD_SPANTYPE /*type*/, void* /*detail*/, void* /*userdata*/);
380 int (*leave_span)(MD_SPANTYPE /*type*/, void* /*detail*/, void* /*userdata*/);
381
382 int (*text)(MD_TEXTTYPE /*type*/, const MD_CHAR* /*text*/, MD_SIZE /*size*/, void* /*userdata*/);
383
384 /* Debug callback. Optional (may be NULL).
385 *
386 * If provided and something goes wrong, this function gets called.
387 * This is intended for debugging and problem diagnosis for developers;
388 * it is not intended to provide any errors suitable for displaying to an
389 * end user.
390 */
391 void (*debug_log)(const char* /*msg*/, void* /*userdata*/);
392
393 /* Reserved. Set to NULL.
394 */
395 void (*syntax)(void);
396 } MD_PARSER;

402 /* Parse the Markdown document stored in the string 'text' of size 'size'.
403 * The parser provides callbacks to be called during the parsing so the
404 * caller can render the document on the screen or convert the Markdown
405 * to another format.
406 *
407 * Zero is returned on success. If a runtime error occurs (e.g. a memory
408 * fails), -1 is returned. If the processing is aborted due any callback
409 * returning non-zero, the return value of the callback is returned.
410 */
411 int md_parse(const MD_CHAR* text, MD_SIZE size, const MD_PARSER* parser, void* userdata);
