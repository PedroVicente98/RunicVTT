188void imgui_md::BLOCK_TABLE(const MD_BLOCK_TABLE_DETAIL*, bool e)
189{
190 if (e)
191 {
192 m_table_row_pos.clear();
193 m_table_col_pos.clear();
194
195 m_table_last_pos = ImGui::GetCursorPos();
196 }
197 else
198 {
199
200 ImGui::NewLine();
201
202 if (m_table_border)
203 {
204
205 m_table_last_pos.y = ImGui::GetCursorPos().y;
206
207 m_table_col_pos.push_back(m_table_last_pos.x);
208 m_table_row_pos.push_back(m_table_last_pos.y);
209
210 const ImVec2 wp = ImGui::GetWindowPos();
211 const ImVec2 sp = ImGui::GetStyle().ItemSpacing;
212 const float wx = wp.x + sp.x / 2;
213 const float wy = wp.y - sp.y / 2 - ImGui::GetScrollY();
214
215 for (int i = 0; i < m_table_col_pos.size(); ++i)
216 {
217 m_table_col_pos[i] += wx;
218 }
219
220 for (int i = 0; i < m_table_row_pos.size(); ++i)
221 {
222 m_table_row_pos[i] += wy;
223 }
224
226
227 const ImColor c = ImGui::GetStyle().Colors[ImGuiCol_TextDisabled];
228
229 ImDrawList* dl = ImGui::GetWindowDrawList();
230
231 const float xmin = m_table_col_pos.front();
232 const float xmax = m_table_col_pos.back();
233 for (int i = 0; i < m_table_row_pos.size(); ++i)
234 {
235 const float p = m_table_row_pos[i];
236 dl->AddLine(ImVec2(xmin, p), ImVec2(xmax, p), c,
237 i == 1 && m_table_header_highlight ? 2.0f : 1.0f);
238 }
239
240 const float ymin = m_table_row_pos.front();
241 const float ymax = m_table_row_pos.back();
242 for (int i = 0; i < m_table_col_pos.size(); ++i)
243 {
244 const float p = m_table_col_pos[i];
245 dl->AddLine(ImVec2(p, ymin), ImVec2(p, ymax), c, 1.0f);
246 }
247 }
248 }
249}

280void imgui_md::BLOCK_TD(const MD_BLOCK_TD_DETAIL*, bool e)

302 const ImVec2 p = ImGui::GetCursorPos();
303 ImGui::Unindent(m_table_col_pos[m_table_next_column - 1]);
304 ImGui::SetCursorPosX(p.x);
305 if (p.y > m_table_last_pos.y)
306 m_table_last_pos.y = p.y;
307 }
308 ImGui::TextUnformatted("");
309
310 if (!m_table_border && e && m_table_next_column == 1)
311 {
312 ImGui::SameLine(0.0f, 0.0f);
313 }
314 else
315 {
316 ImGui::SameLine();

459void imgui_md::render_text(const char* str, const char* str_end)
460{
461 const float scale = ImGui::GetIO().FontGlobalScale;
462 const ImGuiStyle& s = ImGui::GetStyle();
463 bool is_lf = false;
464
465 while (!m_is_image && str < str_end)
466 {
467
468 const char* te = str_end;
469
470 if (!m_is_table_header)
471 {
472
473 float wl = ImGui::GetContentRegionAvail().x;
474
475 if (m_is_table_body)
476 {
477 wl = (m_table_next_column < m_table_col_pos.size() ? m_table_col_pos[m_table_next_column] : m_table_last_pos.x);
478 wl -= ImGui::GetCursorPosX();
479 }
480
481 te = ImGui::GetFont()->CalcWordWrapPositionA(
482 scale, str, str_end, wl);
483
484 if (te == str)
485 ++te;
486 }
487
488 ImGui::TextUnformatted(str, te);
489
490 if (te > str && *(te - 1) == '\n')
491 {
492 is_lf = true;
493 }
494
495 if (!m_href.empty())
496 {
497
498 ImVec4 c;
499 if (ImGui::IsItemHovered())
500 {
501
502 ImGui::SetTooltip("%s", m_href.c_str());
503
504 c = s.Colors[ImGuiCol_ButtonHovered];
505 if (ImGui::IsMouseReleased(0))
506 {
507 open_url();
508 }
509 }
510 else
511 {
512 c = s.Colors[ImGuiCol_Button];
513 }
514 line(c, true);
515 }
516 if (m_is_underline)
517 {
518 line(s.Colors[ImGuiCol_Text], true);
519 }
520 if (m_is_strikethrough)
521 {
522 line(s.Colors[ImGuiCol_Text], false);
523 }
524
525 str = te;
526
527 while (str < str_end && *str == ' ')
528 ++str;
529 }
530
531 if (!is_lf)
532 ImGui::SameLine(0.0f, 0.0f);
533}

547static bool skip_spaces(const std::string& d, size_t& p)
548{
549 for (; p < d.length(); ++p)
550 {
551 if (d[p] != ' ' && d[p] != '\t')
552 {
553 break;
554 }
555 }
556 return p < d.length();
557}
558
559static std::string get_div_class(const char* str, const char* str_end)
560{
561 if (str_end <= str)
562 return "";
563
564 std::string d(str, str_end - str);
565 if (d.back() == '>')
566 d.pop_back();
567
568 const char attr[] = "class";
569 size_t p = d.find(attr);
570 if (p == std::string::npos)
571 return "";
572 p += sizeof(attr) - 1;
573
574 if (!skip_spaces(d, p))
575 return "";
576
577 if (d[p] != '=')
578 return "";
579 ++p;
580
581 if (!skip_spaces(d, p))
582 return "";
583
584 bool has_q = false;
585
586 if (d[p] == '"' || d[p] == '\'')
587 {
588 has_q = true;
589 ++p;
590 }
591 if (p == d.length())
592 return "";
593
594 if (!has_q)
595 {
596 if (!skip_spaces(d, p))
597 return "";
598 }
599
600 size_t pe;
601 for (pe = p; pe < d.length(); ++pe)
602 {
603
604 const char c = d[pe];
605
606 if (has_q)
607 {
608 if (c == '"' || c == '\'')
609 {
610 break;
611 }
612 }
613 else
614 {
615 if (c == ' ' || c == '\t')
616 {
617 break;
618 }
619 }
620 }
621
622 return d.substr(p, pe - p);
623}

625bool imgui_md::check_html(const char* str, const char* str_end)
626{
627 const size_t sz = str_end - str;
628
629 if (strncmp(str, "<br>", sz) == 0)
630 {
631 ImGui::NewLine();
632 return true;
633 }
634 if (strncmp(str, "<hr>", sz) == 0)
635 {
636 ImGui::Separator();
637 return true;
638 }
639 if (strncmp(str, "<u>", sz) == 0)
640 {
641 m_is_underline = true;
642 return true;
643 }
644 if (strncmp(str, "</u>", sz) == 0)
645 {
646 m_is_underline = false;
647 return true;
648 }
649
650 const size_t div_sz = 4;
651 if (strncmp(str, "<div", sz > div_sz ? div_sz : sz) == 0)
652 {
653 m_div_stack.emplace_back(get_div_class(str + div_sz, str_end));
654 html_div(m_div_stack.back(), true);
655 return true;
656 }
657 if (strncmp(str, "</div>", sz) == 0)
658 {
659 if (m_div_stack.empty())
660 return false;
661 html_div(m_div_stack.back(), false);
662 m_div_stack.pop_back();
663 return true;
664 }
665 return false;
666}

685int imgui_md::text(MD_TEXTTYPE type, const char* str, const char* str_end)
686{
687 switch (type)
688 {
689 case MD_TEXT_NORMAL:
690 render_text(str, str_end);
691 break;
692 case MD_TEXT_CODE:
693 render_text(str, str_end);
694 break;
695 case MD_TEXT_NULLCHAR:
696 break;
697 case MD_TEXT_BR:
698 ImGui::NewLine();
699 break;
700 case MD_TEXT_SOFTBR:
701 soft_break();
702 break;
703 case MD_TEXT_ENTITY:
704 if (!render_entity(str, str_end))
705 {
706 render_text(str, str_end);
707 };
708 break;
709 case MD_TEXT_HTML:
710 if (!check_html(str, str_end))
711 {
712 render_text(str, str_end);
713 }
714 break;
715 case MD_TEXT_LATEXMATH:
716 render_text(str, str_end);
717 break;
718 default:
719 break;
720 }
721
722 if (m_is_table_header)
723 {
724 const float x = ImGui::GetCursorPosX();
725 if (x > m_table_last_pos.x)
726 m_table_last_pos.x = x;
727 }
728
729 return 0;
730}

874bool imgui_md::get_image(image_info& nfo) const
875{
876 //Use m_href to identify images
877
878 //Example - Imgui font texture
879 nfo.texture_id = ImGui::GetIO().Fonts->TexID;
